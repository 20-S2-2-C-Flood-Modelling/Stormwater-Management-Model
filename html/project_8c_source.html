111 void project_readInput()
112 //
113 // Input: none
114 // Output: none
115 // Purpose: retrieves project data from input file.
116 //
117 {
118  // --- create hash tables for fast retrieval of objects by ID names
119  createHashTables();
120 
121  // --- count number of objects in input file and create them
122  input_countObjects();
123  createObjects();
124 
125  // --- read project data from input file
126  input_readData();
127  if ( ErrorCode ) return;
128 
129  // --- establish starting & ending date/time
130  StartDateTime = StartDate + StartTime;
131  EndDateTime = EndDate + EndTime;
132  ReportStart = ReportStartDate + ReportStartTime;
133  ReportStart = MAX(ReportStart, StartDateTime);
134 
135  // --- check for valid starting & ending date/times
136  if ( EndDateTime <= StartDateTime )
137  {
138  report_writeErrorMsg(ERR_START_DATE, "");
139  }
140  else if ( EndDateTime <= ReportStart )
141  {
142  report_writeErrorMsg(ERR_REPORT_DATE, "");
143  }
144  else
145  {
148  // --- compute total duration of simulation in seconds
149  TotalDuration = floor((EndDateTime - StartDateTime) * SECperDAY);
150 
151  // --- reporting step must be <= total duration
152  if ( (double)ReportStep > TotalDuration )
153  {
154  ReportStep = (int)(TotalDuration);
155  }
156 
157  // --- reporting step can't be < routing step
158  if ( (double)ReportStep < RouteStep )
159  {
160  report_writeErrorMsg(ERR_REPORT_STEP, "");
161  }
162 
163  // --- convert total duration to milliseconds
164  TotalDuration *= 1000.0;
165  }
167 }

171 void project_validate()
172 //
173 // Input: none
174 // Output: none
175 // Purpose: checks validity of project data.
176 //
177 {
178  int i;
179  int j;
180  int err;
181 
182  // --- validate Curves and TimeSeries
183  for ( i=0; i<Nobjects[CURVE]; i++ )
184  {
185  err = table_validate(&Curve[i]);
186  if ( err ) report_writeErrorMsg(ERR_CURVE_SEQUENCE, Curve[i].ID);
187  }
188  for ( i=0; i<Nobjects[TSERIES]; i++ )
189  {
190  err = table_validate(&Tseries[i]);
191  if ( err ) report_writeTseriesErrorMsg(err, &Tseries[i]);
192  }
193 
194  // --- validate hydrology objects
195  // (NOTE: order is important !!!!)
196  climate_validate();
197  lid_validate();
198  if ( Nobjects[SNOWMELT] == 0 ) IgnoreSnowmelt = TRUE;
199  if ( Nobjects[AQUIFER] == 0 ) IgnoreGwater = TRUE;
200  for ( i=0; i<Nobjects[GAGE]; i++ ) gage_validate(i);
201  for ( i=0; i<Nobjects[AQUIFER]; i++ ) gwater_validateAquifer(i);
202  for ( i=0; i<Nobjects[SUBCATCH]; i++ ) subcatch_validate(i);
203  for ( i=0; i<Nobjects[SNOWMELT]; i++ ) snow_validateSnowmelt(i);
204 
205  // --- compute geometry tables for each shape curve
206  j = 0;
207  for ( i=0; i<Nobjects[CURVE]; i++ )
208  {
209  if ( Curve[i].curveType == SHAPE_CURVE )
210  {
211  Curve[i].refersTo = j;
212  Shape[j].curve = i;
213  if ( !shape_validate(&Shape[j], &Curve[i]) )
214  report_writeErrorMsg(ERR_CURVE_SEQUENCE, Curve[i].ID);
215  j++;
216  }
217  }
218 
219  // --- validate links before nodes, since the latter can
220  // result in adjustment of node depths
221  for ( i=0; i<Nobjects[NODE]; i++) Node[i].oldDepth = Node[i].fullDepth;
222  for ( i=0; i<Nobjects[LINK]; i++) link_validate(i);
223  for ( i=0; i<Nobjects[NODE]; i++) node_validate(i);
224 
225  // --- adjust time steps if necessary
226  if ( DryStep < WetStep )
227  {
228  report_writeWarningMsg(WARN06, "");
229  DryStep = WetStep;
230  }
231  if ( RouteStep > (double)WetStep )
232  {
233  report_writeWarningMsg(WARN07, "");
234  RouteStep = WetStep;
235  }
236 
237  // --- adjust individual reporting flags to match global reporting flag
238  if ( RptFlags.subcatchments == ALL )
239  for (i=0; i<Nobjects[SUBCATCH]; i++) Subcatch[i].rptFlag = TRUE;
240  if ( RptFlags.nodes == ALL )
241  for (i=0; i<Nobjects[NODE]; i++) Node[i].rptFlag = TRUE;
242  if ( RptFlags.links == ALL )
243  for (i=0; i<Nobjects[LINK]; i++) Link[i].rptFlag = TRUE;
244 
245  // --- validate dynamic wave options
246  if ( RouteModel == DW ) dynwave_validate(); //(5.1.008)
247 
248 #pragma omp parallel //(5.1.008)
249 {
250  if ( NumThreads == 0 ) NumThreads = omp_get_num_threads(); //(5.1.008)
251  else NumThreads = MIN(NumThreads, omp_get_num_threads()); //(5.1.008)
252 }
253  if ( Nobjects[LINK] < 4 * NumThreads ) NumThreads = 1; //(5.1.008)
254 
255 }

292 int project_addObject(int type, char *id, int n)
293 //
294 // Input: type = object type
295 // id = object ID string
296 // n = object index
297 // Output: returns 0 if object already added, 1 if not, -1 if hashing fails
298 // Purpose: adds an object ID to a hash table
299 //
300 {
301  int result;
302  int len;
303  char *newID;
304 
305  // --- do nothing if object already placed in hash table
306  if ( project_findObject(type, id) >= 0 ) return 0;
307 
308  // --- use memory from the hash tables' common memory pool to store
309  // a copy of the object's ID string
310  len = strlen(id) + 1;
311  newID = (char *) Alloc(len*sizeof(char));
312  strcpy(newID, id);
313 
314  // --- insert object's ID into the hash table for that type of object
315  result = HTinsert(Htable[type], newID, n);
316  if ( result == 0 ) result = -1;
317  return result;
318 }

351 double ** project_createMatrix(int nrows, int ncols)
352 //
353 // Input: nrows = number of rows (0-based)
354 // ncols = number of columns (0-based)
355 // Output: returns a pointer to a matrix
356 // Purpose: allocates memory for a matrix of doubles.
357 //
358 {
359  int i,j;
360  double **a;
361 
362  // --- allocate pointers to rows
363  a = (double **) malloc(nrows * sizeof(double *));
364  if ( !a ) return NULL;
365 
366  // --- allocate rows and set pointers to them
367  a[0] = (double *) malloc (nrows * ncols * sizeof(double));
368  if ( !a[0] ) return NULL;
369  for ( i = 1; i < nrows; i++ ) a[i] = a[i-1] + ncols;
370 
371  for ( i = 0; i < nrows; i++)
372  {
373  for ( j = 0; j < ncols; j++) a[i][j] = 0.0;
374  }
375 
376  // --- return pointer to array of pointers to rows
377  return a;
378 }

398 int project_readOption(char* s1, char* s2)
399 //
400 // Input: s1 = option keyword
401 // s2 = string representation of option's value
402 // Output: returns error code
403 // Purpose: reads a project option from a pair of string tokens.
404 //
405 // NOTE: all project options have default values assigned in setDefaults().
406 //
407 {
408  int k, m, h, s;
409  double tStep;
410  char strDate[25];
411  DateTime aTime;
412  DateTime aDate;
413 
414  // --- determine which option is being read
415  k = findmatch(s1, OptionWords);
416  if ( k < 0 ) return error_setInpError(ERR_KEYWORD, s1);
417  switch ( k )
418  {
419  // --- choice of flow units
420  case FLOW_UNITS:
421  m = findmatch(s2, FlowUnitWords);
422  if ( m < 0 ) return error_setInpError(ERR_KEYWORD, s2);
423  FlowUnits = m;
424  if ( FlowUnits <= MGD ) UnitSystem = US;
425  else UnitSystem = SI;
426  break;
427 
428  // --- choice of infiltration modeling method
429  case INFIL_MODEL:
430  m = findmatch(s2, InfilModelWords);
431  if ( m < 0 ) return error_setInpError(ERR_KEYWORD, s2);
432  InfilModel = m;
433  break;
434 
435  // --- choice of flow routing method
436  case ROUTE_MODEL:
437  m = findmatch(s2, RouteModelWords);
438  if ( m < 0 ) m = findmatch(s2, OldRouteModelWords);
439  if ( m < 0 ) return error_setInpError(ERR_KEYWORD, s2);
440  if ( m == NO_ROUTING ) IgnoreRouting = TRUE;
441  else RouteModel = m;
442  if ( RouteModel == EKW ) RouteModel = KW;
443  break;
444 
445  // --- simulation start date
446  case START_DATE:
447  if ( !datetime_strToDate(s2, &StartDate) )
448  {
449  return error_setInpError(ERR_DATETIME, s2);
450  }
451  break;
452 
453  // --- simulation start time of day
454  case START_TIME:
455  if ( !datetime_strToTime(s2, &StartTime) )
456  {
457  return error_setInpError(ERR_DATETIME, s2);
458  }
459  break;
460 
461  // --- simulation ending date
462  case END_DATE:
463  if ( !datetime_strToDate(s2, &EndDate) )
464  {
465  return error_setInpError(ERR_DATETIME, s2);
466  }
467  break;
468 
469  // --- simulation ending time of day
470  case END_TIME:
471  if ( !datetime_strToTime(s2, &EndTime) )
472  {
473  return error_setInpError(ERR_DATETIME, s2);
474  }
475  break;
476 
477  // --- reporting start date
478  case REPORT_START_DATE:
479  if ( !datetime_strToDate(s2, &ReportStartDate) )
480  {
481  return error_setInpError(ERR_DATETIME, s2);
482  }
483  break;
484 
485  // --- reporting start time of day
486  case REPORT_START_TIME:
487  if ( !datetime_strToTime(s2, &ReportStartTime) )
488  {
489  return error_setInpError(ERR_DATETIME, s2);
490  }
491  break;
492 
493  // --- day of year when street sweeping begins or when it ends
494  // (year is arbitrarily set to 1947 so that the dayOfYear
495  // function can be applied)
496  case SWEEP_START:
497  case SWEEP_END:
498  strcpy(strDate, s2);
499  strcat(strDate, "/1947");
500  if ( !datetime_strToDate(strDate, &aDate) )
501  {
502  return error_setInpError(ERR_DATETIME, s2);
503  }
504  m = datetime_dayOfYear(aDate);
505  if ( k == SWEEP_START ) SweepStart = m;
506  else SweepEnd = m;
507  break;
508 
509  // --- number of antecedent dry days
510  case START_DRY_DAYS:
511  StartDryDays = atof(s2);
512  if ( StartDryDays < 0.0 )
513  {
514  return error_setInpError(ERR_NUMBER, s2);
515  }
516  break;
517 
518  // --- runoff or reporting time steps
519  // (input is in hrs:min:sec format, time step saved as seconds)
520  case WET_STEP:
521  case DRY_STEP:
522  case REPORT_STEP:
523  if ( !datetime_strToTime(s2, &aTime) )
524  {
525  return error_setInpError(ERR_DATETIME, s2);
526  }
527  datetime_decodeTime(aTime, &h, &m, &s);
528  h += 24*(int)aTime;
529  s = s + 60*m + 3600*h;
530  if ( s <= 0 ) return error_setInpError(ERR_NUMBER, s2);
531  switch ( k )
532  {
533  case WET_STEP: WetStep = s; break;
534  case DRY_STEP: DryStep = s; break;
535  case REPORT_STEP: ReportStep = s; break;
536  }
537  break;
538 
539  // --- type of damping applied to inertial terms of dynamic wave routing
540  case INERT_DAMPING:
541  m = findmatch(s2, InertDampingWords);
542  if ( m < 0 ) return error_setInpError(ERR_KEYWORD, s2);
543  else InertDamping = m;
544  break;
545 
546  // --- Yes/No options (NO = 0, YES = 1)
547  case ALLOW_PONDING:
548  case SLOPE_WEIGHTING:
549  case SKIP_STEADY_STATE:
550  case IGNORE_RAINFALL:
551  case IGNORE_SNOWMELT:
552  case IGNORE_GWATER:
553  case IGNORE_ROUTING:
554  case IGNORE_QUALITY:
555  case IGNORE_RDII: //(5.1.004)
556  m = findmatch(s2, NoYesWords);
557  if ( m < 0 ) return error_setInpError(ERR_KEYWORD, s2);
558  switch ( k )
559  {
560  case ALLOW_PONDING: AllowPonding = m; break;
561  case SLOPE_WEIGHTING: SlopeWeighting = m; break;
562  case SKIP_STEADY_STATE: SkipSteadyState = m; break;
563  case IGNORE_RAINFALL: IgnoreRainfall = m; break;
564  case IGNORE_SNOWMELT: IgnoreSnowmelt = m; break;
565  case IGNORE_GWATER: IgnoreGwater = m; break;
566  case IGNORE_ROUTING: IgnoreRouting = m; break;
567  case IGNORE_QUALITY: IgnoreQuality = m; break;
568  case IGNORE_RDII: IgnoreRDII = m; break; //(5.1.004)
569  }
570  break;
571 
572  case NORMAL_FLOW_LTD:
573  m = findmatch(s2, NormalFlowWords);
574  if ( m < 0 ) m = findmatch(s2, NoYesWords);
575  if ( m < 0 ) return error_setInpError(ERR_KEYWORD, s2);
576  NormalFlowLtd = m;
577  break;
578 
579  case FORCE_MAIN_EQN:
580  m = findmatch(s2, ForceMainEqnWords);
581  if ( m < 0 ) return error_setInpError(ERR_KEYWORD, s2);
582  ForceMainEqn = m;
583  break;
584 
585  case LINK_OFFSETS:
586  m = findmatch(s2, LinkOffsetWords);
587  if ( m < 0 ) return error_setInpError(ERR_KEYWORD, s2);
588  LinkOffsets = m;
589  break;
590 
591  // --- compatibility option for selecting solution method for
592  // dynamic wave flow routing (NOT CURRENTLY USED)
593  case COMPATIBILITY:
594  if ( strcomp(s2, "3") ) Compatibility = SWMM3;
595  else if ( strcomp(s2, "4") ) Compatibility = SWMM4;
596  else if ( strcomp(s2, "5") ) Compatibility = SWMM5;
597  else return error_setInpError(ERR_KEYWORD, s2);
598  break;
599 
600  // --- routing or lengthening time step (in decimal seconds)
601  // (lengthening time step is used in Courant stability formula
602  // to artificially lengthen conduits for dynamic wave flow routing
603  // (a value of 0 means that no lengthening is used))
604  case ROUTE_STEP:
605  case LENGTHENING_STEP:
606  if ( !getDouble(s2, &tStep) )
607  {
608  if ( !datetime_strToTime(s2, &aTime) )
609  {
610  return error_setInpError(ERR_NUMBER, s2);
611  }
612  else
613  {
614  datetime_decodeTime(aTime, &h, &m, &s);
615  h += 24*(int)aTime;
616  s = s + 60*m + 3600*h;
617  tStep = s;
618  }
619  }
620  if ( k == ROUTE_STEP )
621  {
622  if ( tStep <= 0.0 ) return error_setInpError(ERR_NUMBER, s2);
623  RouteStep = tStep;
624  }
625  else LengtheningStep = MAX(0.0, tStep);
626  break;
627 
629 
630  // --- minimum variable time step for dynamic wave routing
631  case MIN_ROUTE_STEP:
632  if ( !getDouble(s2, &MinRouteStep) || MinRouteStep < 0.0 )
633  return error_setInpError(ERR_NUMBER, s2);
634  break;
635 
636  case NUM_THREADS:
637  m = atoi(s2);
638  if ( m < 0 ) return error_setInpError(ERR_NUMBER, s2);
639  NumThreads = m;
640  break;
642 
643  // --- safety factor applied to variable time step estimates under
644  // dynamic wave flow routing (value of 0 indicates that variable
645  // time step option not used)
646  case VARIABLE_STEP:
647  if ( !getDouble(s2, &CourantFactor) )
648  return error_setInpError(ERR_NUMBER, s2);
649  if ( CourantFactor < 0.0 || CourantFactor > 2.0 )
650  return error_setInpError(ERR_NUMBER, s2);
651  break;
652 
653  // --- minimum surface area (ft2 or sq. meters) associated with nodes
654  // under dynamic wave flow routing
655  case MIN_SURFAREA:
656  MinSurfArea = atof(s2);
657  break;
658 
659  // --- minimum conduit slope (%)
660  case MIN_SLOPE:
661  if ( !getDouble(s2, &MinSlope) )
662  return error_setInpError(ERR_NUMBER, s2);
663  if ( MinSlope < 0.0 || MinSlope >= 100 )
664  return error_setInpError(ERR_NUMBER, s2);
665  MinSlope /= 100.0;
666  break;
667 
668  // --- maximum trials / time step for dynamic wave routing
669  case MAX_TRIALS:
670  m = atoi(s2);
671  if ( m < 0 ) return error_setInpError(ERR_NUMBER, s2);
672  MaxTrials = m;
673  break;
674 
675  // --- head convergence tolerance for dynamic wave routing
676  case HEAD_TOL:
677  if ( !getDouble(s2, &HeadTol) )
678  {
679  return error_setInpError(ERR_NUMBER, s2);
680  }
681  break;
682 
683  // --- steady state tolerance on system inflow - outflow
684  case SYS_FLOW_TOL:
685  if ( !getDouble(s2, &SysFlowTol) )
686  {
687  return error_setInpError(ERR_NUMBER, s2);
688  }
689  SysFlowTol /= 100.0;
690  break;
691 
692  // --- steady state tolerance on nodal lateral inflow
693  case LAT_FLOW_TOL:
694  if ( !getDouble(s2, &LatFlowTol) )
695  {
696  return error_setInpError(ERR_NUMBER, s2);
697  }
698  LatFlowTol /= 100.0;
699  break;
700 
701  case TEMPDIR: // Temporary Directory
702  sstrncpy(TempDir, s2, MAXFNAME);
703  break;

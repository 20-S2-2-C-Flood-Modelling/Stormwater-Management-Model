63 int iface_readFileParams(char* tok[], int ntoks)
64 //
65 // Input: tok[] = array of string tokens
66 // ntoks = number of tokens
67 // Output: returns an error code
68 // Purpose: reads interface file information from a line of input data.
69 //
70 // Data format is:
71 // USE/SAVE FileType FileName
72 //
73 {
74  char k;
75  int j;
76 
77  // --- determine file disposition and type
78  if ( ntoks < 2 ) return error_setInpError(ERR_ITEMS, "");
79  k = (char)findmatch(tok[0], FileModeWords);
80  if ( k < 0 ) return error_setInpError(ERR_KEYWORD, tok[0]);
81  j = findmatch(tok[1], FileTypeWords);
82  if ( j < 0 ) return error_setInpError(ERR_KEYWORD, tok[1]);
83  if ( ntoks < 3 ) return 0;
84 
85  // --- process file name
86  switch ( j )
87  {
88  case RAINFALL_FILE:
89  Frain.mode = k;
90  sstrncpy(Frain.name, tok[2], MAXFNAME);
91  break;
92 
93  case RUNOFF_FILE:
94  Frunoff.mode = k;
95  sstrncpy(Frunoff.name, tok[2], MAXFNAME);
96  break;
97 
98  case HOTSTART_FILE:
99  if ( k == USE_FILE )
100  {
101  Fhotstart1.mode = k;
102  sstrncpy(Fhotstart1.name, tok[2], MAXFNAME);
103  }
104  else if ( k == SAVE_FILE )
105  {
106  Fhotstart2.mode = k;
107  sstrncpy(Fhotstart2.name, tok[2], MAXFNAME);
108  }
109  break;
110 
111  case RDII_FILE:
112  Frdii.mode = k;
113  sstrncpy(Frdii.name, tok[2], MAXFNAME);
114  break;
115 
116  case INFLOWS_FILE:
117  if ( k != USE_FILE ) return error_setInpError(ERR_ITEMS, "");
118  Finflows.mode = k;
119  sstrncpy(Finflows.name, tok[2], MAXFNAME);
120  break;
121 
122  case OUTFLOWS_FILE:
123  if ( k != SAVE_FILE ) return error_setInpError(ERR_ITEMS, "");
124  Foutflows.mode = k;
125  sstrncpy(Foutflows.name, tok[2], MAXFNAME);
126  break;
127  }
128  return 0;
129 }

182 int iface_getNumIfaceNodes(DateTime currentDate)
183 //
184 // Input: currentDate = current date/time
185 // Output: returns number of interface nodes if data exists or
186 // 0 otherwise
187 // Purpose: reads inflow data from interface file at current date.
188 //
189 {
190  // --- return 0 if file begins after current date
191  if ( OldIfaceDate > currentDate ) return 0;
192 
193  // --- keep updating new interface values until current date bracketed
194  while ( NewIfaceDate < currentDate && NewIfaceDate != NO_DATE )
195  {
196  setOldIfaceValues();
197  readNewIfaceValues();
198  }
199 
200  // --- return 0 if no data available
201  if ( NewIfaceDate == NO_DATE ) return 0;
202 
203  // --- find fraction current date is bewteen old & new interface dates
204  IfaceFrac = (currentDate - OldIfaceDate) / (NewIfaceDate - OldIfaceDate);
205  IfaceFrac = MAX(0.0, IfaceFrac);
206  IfaceFrac = MIN(IfaceFrac, 1.0);
207 
208  // --- return number of interface nodes
209  return NumIfaceNodes;
210 }

227 double iface_getIfaceFlow(int index)
228 //
229 // Input: index = interface file node index
230 // Output: returns inflow to node
231 // Purpose: finds interface flow for particular node index.
232 //
233 {
234  double q1, q2;
235 
236  if ( index >= 0 && index < NumIfaceNodes )
237  {
238  // --- interpolate flow between old and new values
239  q1 = OldIfaceValues[index][0];
240  q2 = NewIfaceValues[index][0];
241  return (1.0 - IfaceFrac)*q1 + IfaceFrac*q2;
242  }
243  else return 0.0;
244 }

248 double iface_getIfaceQual(int index, int pollut)
249 //
250 // Input: index = index of node on interface file
251 // pollut = index of pollutant on interface file
252 // Output: returns inflow pollutant concentration
253 // Purpose: finds interface concentration for particular node index & pollutant.
254 //
255 {
256  int j;
257  double c1, c2;
258 
259  if ( index >= 0 && index < NumIfaceNodes )
260  {
261  // --- find index of pollut on interface file
262  j = IfacePolluts[pollut];
263  if ( j < 0 ) return 0.0;
264 
265  // --- interpolate flow between old and new values
266  // (remember that 1st col. of values matrix is for flow)
267  c1 = OldIfaceValues[index][j+1];
268  c2 = NewIfaceValues[index][j+1];
269  return (1.0 - IfaceFrac)*c1 + IfaceFrac*c2;
270  }
271  else return 0.0;
272 }

447 int getIfaceFilePolluts()
448 //
449 // Input: none
450 // Output: returns an error code
451 // Purpose: reads names of pollutants saved on the inflows interface file.
452 //
453 {
454  int i, j;
455  char line[MAXLINE+1]; // line from inflows interface file
456  char s1[MAXLINE+1]; // general string variable
457  char s2[MAXLINE+1];
458 
459  // --- read number of pollutants (minus FLOW)
460  fgets(line, MAXLINE, Finflows.file);
461  sscanf(line, "%d", &NumIfacePolluts);
462  NumIfacePolluts--;
463  if ( NumIfacePolluts < 0 ) return ERR_ROUTING_FILE_FORMAT;
464 
465  // --- read flow units
466  fgets(line, MAXLINE, Finflows.file);
467  sscanf(line, "%s %s", s1, s2);
468  if ( !strcomp(s1, "FLOW") ) return ERR_ROUTING_FILE_FORMAT;
469  IfaceFlowUnits = findmatch(s2, FlowUnitWords);
470  if ( IfaceFlowUnits < 0 ) return ERR_ROUTING_FILE_FORMAT;
471 
472  // --- allocate memory for pollutant index array
473  if ( Nobjects[POLLUT] > 0 )
474  {
475  IfacePolluts = (int *) calloc(Nobjects[POLLUT], sizeof(int));
476  if ( !IfacePolluts ) return ERR_MEMORY;
477  for (i=0; i<Nobjects[POLLUT]; i++) IfacePolluts[i] = -1;
478  }
479 
480  // --- read pollutant names & units
481  if ( NumIfacePolluts > 0 )
482  {
483  // --- check each pollutant name on file with project's pollutants
484  for (i=0; i<NumIfacePolluts; i++)
485  {
486  if ( feof(Finflows.file) ) return ERR_ROUTING_FILE_FORMAT;
487  fgets(line, MAXLINE, Finflows.file);
488  sscanf(line, "%s %s", s1, s2);
489  if ( Nobjects[POLLUT] > 0 )
490  {
491  j = project_findObject(POLLUT, s1);
492  if ( j < 0 ) continue;
493  if ( !strcomp(s2, QualUnitsWords[Pollut[j].units]) )
494  return ERR_ROUTING_FILE_NOMATCH;
495  IfacePolluts[j] = i;
496  }
497  }
498  }
499  return 0;
500 }

541 void readNewIfaceValues()
542 //
543 // Input: none
544 // Output: none
545 // Purpose: reads data from inflows interface file for next date.
546 //
547 {
548  int i, j;
549  char* s;
550  int yr = 0, mon = 0, day = 0,
551  hr = 0, min = 0, sec = 0; // year, month, day, hour, minute, second
552  char line[MAXLINE+1]; // line from interface file
553 
554  // --- read a line for each interface node
555  NewIfaceDate = NO_DATE;
556  for (i=0; i<NumIfaceNodes; i++)
557  {
558  if ( feof(Finflows.file) ) return;
559  fgets(line, MAXLINE, Finflows.file);
560 
561  // --- parse date & time from line
562  if ( strtok(line, SEPSTR) == NULL ) return;
563  s = strtok(NULL, SEPSTR);
564  if ( s == NULL ) return;
565  yr = atoi(s);
566  s = strtok(NULL, SEPSTR);
567  if ( s == NULL ) return;
568  mon = atoi(s);
569  s = strtok(NULL, SEPSTR);
570  if ( s == NULL ) return;
571  day = atoi(s);
572  s = strtok(NULL, SEPSTR);
573  if ( s == NULL ) return;
574  hr = atoi(s);
575  s = strtok(NULL, SEPSTR);
576  if ( s == NULL ) return;
577  min = atoi(s);
578  s = strtok(NULL, SEPSTR);
579  if ( s == NULL ) return;
580  sec = atoi(s);
581 
582  // --- parse flow value
583  s = strtok(NULL, SEPSTR);
584  if ( s == NULL ) return;
585  NewIfaceValues[i][0] = atof(s) / Qcf[IfaceFlowUnits];
586 
587  // --- parse pollutant values
588  for (j=1; j<=NumIfacePolluts; j++)
589  {
590  s = strtok(NULL, SEPSTR);
591  if ( s == NULL ) return;
592  NewIfaceValues[i][j] = atof(s);
593  }
594 
595  }
596 
597  // --- encode date & time values
598  NewIfaceDate = datetime_encodeDate(yr, mon, day) +
599  datetime_encodeTime(hr, min, sec);
600 }

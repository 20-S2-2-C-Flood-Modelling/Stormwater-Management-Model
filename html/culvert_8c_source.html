170 double culvert_getInflow(int j, double q0, double h)
171 //
172 // Input: j = link index
173 // q0 = unmodified flow rate (cfs)
174 // h = upstream head (ft)
175 // Output: returns modified flow rate through culvert (cfs)
176 // Purpose: uses FHWA HEC-5 equations to find flow through inlet
177 // controlled culverts
178 //
179 {
180  int code, //culvert type code number
181  k, //conduit index
182  condition; //flow condition
183  double y, //current depth (ft)
184  y1, //unsubmerged depth limit (ft)
185  y2, //submerged depth limit (ft)
186  q; //inlet-controlled flow (cfs)
187  TCulvert culvert; //intermediate results
188 
189  // --- check that we have a culvert conduit
190  if ( Link[j].type != CONDUIT ) return q0;
191  culvert.xsect = &Link[j].xsect;
192  code = culvert.xsect->culvertCode;
193  if ( code <= 0 || code > MAX_CULVERT_CODE ) return q0;
194 
195  // --- compute often-used variables
196  k = Link[j].subIndex;
197  culvert.yFull = culvert.xsect->yFull;
198  culvert.ad = culvert.xsect->aFull * sqrt(culvert.yFull);
199 
200  // --- slope correction factor (-7 for mitered inlets, 0.5 for others)
201  switch (code)
202  {
203  case 5:
204  case 37:
205  case 46: culvert.scf = -7.0 * Conduit[k].slope; break;
206  default: culvert.scf = 0.5 * Conduit[k].slope;
207  }
208 
209  // --- find head relative to culvert's upstream invert
210  // (can be greater than yFull when inlet is submerged)
211  y = h - (Node[Link[j].node1].invertElev + Link[j].offset1);
212 
213  // --- check for submerged flow (based on FHWA criteria of Q/AD > 4)
214  y2 = culvert.yFull * (16.0 * Params[code][C] + Params[code][Y] - culvert.scf);
215  if ( y >= y2 )
216  {
217  q = getSubmergedFlow(code, y, &culvert);
218  condition = 2;
219  }
220  else
221  {
222  // --- check for unsubmerged flow (based on arbitrary limit of 0.95 full)
223  y1 = 0.95 * culvert.yFull;
224  if ( y <= y1 )
225  {
226  q = getUnsubmergedFlow(code, y, &culvert);
227  condition = 1;
228  }
229  // --- flow is in transition zone
230  else
231  {
232  q = getTransitionFlow(code, y, y1, y2, &culvert);
233  condition = 0;
234  }
235  }
236 
237  // --- check if inlet controls and replace conduit's value of dq/dh
238  if ( q < q0 )
239  {
240  // --- for debugging only
241  //if ( RptFlags.controls ) report_CulvertControl(j, q0, q, condition,
242  // y / culvert.yFull);
243 
244  Link[j].inletControl = TRUE;
245  Link[j].dqdh = culvert.dQdH;
246  return q;
247  }
248  else return q0;
249 }

253 double getUnsubmergedFlow(int code, double h, TCulvert* culvert)
254 //
255 // Input: code = culvert type code number
256 // h = inlet water depth above culvert invert
257 // culvert = pointer to a culvert data structure
258 // Output: returns flow rate;
259 // computes value of variable Dqdh
260 // Purpose: computes flow rate and its derivative for unsubmerged
261 // culvert inlet.
262 //
263 {
264  double arg;
265  double q;
266 
267  // --- assign shared variables
268  culvert->kk = Params[code][K];
269  culvert->mm = Params[code][M];
270  arg = h / culvert->yFull / culvert->kk;
271 
272  // --- evaluate correct equation form
273  if ( Params[code][FORM] == 1.0)
274  {
275  q = getForm1Flow(h, culvert);
276  }
277  else q = culvert->ad * pow(arg, 1.0/culvert->mm);
278  culvert->dQdH = q / h / culvert->mm;
279  return q;
280 }

284 double getSubmergedFlow(int code, double h, TCulvert* culvert)
285 //
286 // Input: code = culvert type code number
287 // h = inlet head (ft)
288 // culvert = pointer to a culvert data structure
289 // Output: returns flow rate;
290 // computes value of Dqdh
291 // Purpose: computes flow rate and its derivative for submerged
292 // culvert inlet.
293 //
294 {
295  double cc = Params[code][C];
296  double yy = Params[code][Y];
297  double arg = (h/culvert->yFull - yy + culvert->scf) / cc ;
298  double q;
299 
300  if ( arg <= 0.0 )
301  {
302  culvert->dQdH = 0.0;
303  return BIG;
304  }
305  q = sqrt(arg) * culvert->ad;
306  culvert->dQdH = 0.5 * q / arg / culvert->yFull / cc;
307  return q;
308 }

312 double getTransitionFlow(int code, double h, double h1, double h2, TCulvert* culvert)
313 //
314 // Input: code = culvert type code number
315 // h = inlet water depth above culvert invert (ft)
316 // h1 = head limit for unsubmerged condition (ft)
317 // h2 = head limit for submerged condition (ft)
318 // culvert = pointer to a culvert data structure
319 // Output: returns flow rate )cfs);
320 // computes value of Dqdh (cfs/ft)
321 // Purpose: computes flow rate and its derivative for inlet-controlled flow
322 // when inlet water depth lies in the transition range between
323 // submerged and unsubmerged conditions.
324 //
325 {
326  double q1 = getUnsubmergedFlow(code, h1, culvert);
327  double q2 = getSubmergedFlow(code, h2, culvert);
328  double q = q1 + (q2 - q1) * (h - h1) / (h2 - h1);
329  culvert->dQdH = (q2 - q1) / (h2 - h1);
330  return q;
331 }

335 double getForm1Flow(double h, TCulvert* culvert)
336 //
337 // Input: h = inlet water depth above culvert invert
338 // culvert = pointer to a culvert data structure
339 // Output: returns inlet controlled flow rate
340 // Purpose: computes inlet-controlled flow rate for unsubmerged culvert
341 // using FHWA Equation Form1.
342 //
343 // See pages 195-196 of FHWA HEC-5 (2001) for details.
344 //
345 {
346  // --- save re-used terms in culvert structure
347  culvert->hPlus = h / culvert->yFull + culvert->scf;
348 
349  // --- use Ridder's method to solve Equation Form 1 for critical depth
350  // between a range of 0.01h and h
351  findroot_Ridder(0.01*h, h, 0.001, form1Eqn, culvert);
352 
353  // --- return the flow value used in evaluating Equation Form 1
354  return culvert->qc;
355 }

359 double form1Eqn(double yc, void* p)
360 //
361 // Input: yc = critical depth
362 // p = pointer to a TCulvert object
363 // Output: returns residual error
364 // Purpose: evaluates the error in satisfying FHWA culvert Equation Form1:
365 //
366 // h/yFull + 0.5*s = yc/yFull + yh/2/yFull + K[ac/aFull*sqrt(g*yh/yFull)]^M
367 //
368 // for a given value of critical depth yc where:
369 // h = inlet depth above culvert invert
370 // s = culvert slope
371 // yFull = full depth of culvert
372 // yh = hydraulic depth at critical depth
373 // ac = flow area at critical depth
374 // g = accel. of gravity
375 // K and M = coefficients
376 //
377 {
378  double ac, wc, yh;
379  TCulvert* culvert = (TCulvert *)p;
380 
381  ac = xsect_getAofY(culvert->xsect, yc);
382  wc = xsect_getWofY(culvert->xsect, yc);
383  yh = ac/wc;
384 
385  culvert->qc = ac * sqrt(GRAVITY * yh);
386  return culvert->hPlus - yc/culvert->yFull - yh/2.0/culvert->yFull -
387  culvert->kk * pow(culvert->qc/culvert->ad, culvert->mm);
388 }

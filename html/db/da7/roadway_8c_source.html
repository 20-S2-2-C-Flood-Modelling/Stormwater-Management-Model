77 double roadway_getInflow(int j, // link index
78  double dir, // flow direction (+1 or -1)
79  double hRoad, // road elev. (ft)
80  double h1, // upstream head (ft)
81  double h2) // downstream head (ft)
82 {
83  int k; // weir array index
84  int roadSurf; // type of road surface
85  int useVariableCd; // true if HDS-5 coeff. curves used
86  double length, // length of roadway segment transverse to flow (ft)
87  roadWidth, // width of roadway which receives flow (ft)
88  hWr, // water elevation on upstream side of roadway (ft)
89  ht, // water elevation on downstream side of roadway (ft)
90  cD, // discharge coefficient for cfs flow units
91  q = 0.0, // flow across roadway (cfs)
92  dqdh = 0.0; // derivative of flow w.r.t. head (ft2/sec)
93 
94  // --- get road width & surface type
95  if ( Link[j].type != WEIR ) return 0.0;
96  k = Link[j].subIndex;
97  roadWidth = Weir[k].roadWidth;
98  roadSurf = Weir[k].roadSurface;
99 
100  // --- user-supplied discharge coeff.
101  cD = Weir[k].cDisch1;
102  if ( UnitSystem == SI ) cD = cD / 0.552;
103 
104  // --- check if there's enough info to use a variable cD value
105  useVariableCd = FALSE;
106  if ( roadWidth > 0.0 && roadSurf >= 1 ) useVariableCd = TRUE;
107 
108  // --- upstream and downstream heads
109  hWr = h1 - hRoad;
110  ht = h2 - hRoad;
111  if ( hWr > FUDGE )
112  {
113  // --- get discharge coeff. as function of heads
114  if ( useVariableCd ) cD = getCd(hWr, ht, roadWidth, roadSurf);
115 
116  // --- use user-supplied weir length
117  length = Link[j].xsect.wMax;
118 
119  // --- weir eqn. for discharge across roadway
120  q = cD * length * pow(hWr, 1.5);
121  dqdh = 1.5 * q / hWr;
122  }
123 
124  // --- assign output values
125  Link[j].dqdh = dqdh;
126  Link[j].newDepth = MAX(h1 - hRoad, 0.0);
127  Link[j].flowClass = SUBCRITICAL;
128  if ( hRoad > h2 )
129  {
130  if ( dir == 1.0 ) Link[j].flowClass = DN_CRITICAL;
131  else Link[j].flowClass = UP_CRITICAL;
132  }
133  return dir * q;
134 }

138 double getCd(double hWr, double ht, double roadWidth, int roadSurf)
139 {
140  double kT = 1.0; // submergence factor
141  double hL, // ratio of water elevation to road width
142  htH, // ratio of downstream to upstream water depth
143  cR; // roadway discharge coeff.
144 
145  if ( hWr <= 0.0 ) return 0.0;
146  hL = hWr / roadWidth;
147  if ( hL <= 0.15 )
148  {
149  if ( roadSurf == PAVED ) cR = getY(hWr, Cr_Low_Paved, N_Cr_Low_Paved);
150  else cR = getY(hWr, Cr_Low_Gravel, N_Cr_Low_Gravel);
151  }
152  else
153  {
154  if ( roadSurf == PAVED ) cR = getY(hL, Cr_High_Paved, N_Cr_High_Paved);
155  else cR = getY(hL, Cr_High_Gravel, N_Cr_High_Gravel);
156  }
157  if ( ht > 0.0 )
158  {
159  htH = ht / hWr;
160  if ( roadSurf == PAVED ) kT = getY(htH, Kt_Paved, N_Kt_Paved);
161  else kT = getY(htH, Kt_Gravel, N_Kt_Gravel);
162  }
163  return cR * kT;
164 }

168 double getY(double x, const double table[][2], const int n)
169 {
170  int i;
171  double x1, y1,dx, dy;
172  if ( x <= table[0][0] ) return table[0][1];
173  if ( x >= table[n-1][0] ) return table[n-1][1];
174  for (i = 1; i < n; i++)
175  {
176  if ( x <= table[i][0] )
177  {
178  x1 = table[i-1][0];
179  dx = table[i][0] - x1;
180  y1 = table[i-1][1];
181  dy = table[i][1] - y1;
182  return y1 + (x - x1) * dy / dx;
183  }
184  }
185  return table[n-1][1];
186 }

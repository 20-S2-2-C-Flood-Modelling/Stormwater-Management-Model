32 int inflow_readExtInflow(char* tok[], int ntoks)
33 //
34 // Input: tok[] = array of string tokens
35 // ntoks = number of tokens
36 // Output: returns an error message
37 // Purpose: reads parameters of a direct external inflow from a line of input.
38 //
39 // Formats of data line are:
40 // nodeID FLOW tSeriesID (FLOW 1.0 scaleFactor baseline basePat)
41 // nodeID pollutID tSeriesID (CONCEN/MASS unitsFactor scaleFactor baseline basePat)
42 //
43 {
44  int j; // object index
45  int param; // FLOW (-1) or pollutant index
46  int type = CONCEN_INFLOW; // FLOW, CONCEN or MASS inflow
47  int tseries = -1; // time series index
48  int basePat = -1; // baseline pattern
49  double cf = 1.0; // units conversion factor
50  double sf = 1.0; // scaling factor
51  double baseline = 0.0; // baseline value
52 
53  // --- find index of node receiving the inflow
54  if ( ntoks < 3 ) return error_setInpError(ERR_ITEMS, "");
55  j = project_findObject(NODE, tok[0]);
56  if ( j < 0 ) return error_setInpError(ERR_NAME, tok[0]);
57 
58  // --- find index of inflow pollutant or use -1 for FLOW
59  param = project_findObject(POLLUT, tok[1]);
60  if ( param < 0 )
61  {
62  if ( match(tok[1], w_FLOW) ) param = -1;
63  else return error_setInpError(ERR_NAME, tok[1]);
64  }
65 
66  // --- find index of inflow time series (if supplied) in data base
67  if ( strlen(tok[2]) > 0 )
68  {
69  tseries = project_findObject(TSERIES, tok[2]);
70  if ( tseries < 0 ) return error_setInpError(ERR_NAME, tok[2]);
71  Tseries[tseries].refersTo = EXTERNAL_INFLOW;
72  }
73 
74  // --- assign type & cf values for a FLOW inflow
75  if (param == -1)
76  {
77  type = FLOW_INFLOW;
78  }
79 
80  // --- do the same for a pollutant inflow
81  if ( ntoks >= 4 && param > -1)
82  {
83  if ( match(tok[3], w_CONCEN) ) type = CONCEN_INFLOW;
84  else if ( match(tok[3], w_MASS) ) type = MASS_INFLOW;
85  else return error_setInpError(ERR_KEYWORD, tok[3]);
86  if ( ntoks >= 5 && type == MASS_INFLOW )
87  {
88  if ( ! getDouble(tok[4], &cf) )
89  {
90  return error_setInpError(ERR_NUMBER, tok[4]);
91  }
92  if ( cf <= 0.0 ) return error_setInpError(ERR_NUMBER, tok[4]);
93  }
94  }
95 
96  // --- get sf and baseline values
97  if ( ntoks >= 6 )
98  {
99  if ( ! getDouble(tok[5], &sf) )
100  {
101  return error_setInpError(ERR_NUMBER, tok[5]);
102  }
103  }
104  if ( ntoks >= 7 )
105  {
106  if ( ! getDouble(tok[6], &baseline) )
107  {
108  return error_setInpError(ERR_NUMBER, tok[6]);
109  }
110  }
111 
112  // --- get baseline time pattern
113  if ( ntoks >= 8 )
114  {
115  basePat = project_findObject(TIMEPATTERN, tok[7]);
116  if ( basePat < 0 ) return error_setInpError(ERR_NAME, tok[7]);
117  }
118 
119  return(inflow_setExtInflow(j, param, type, tseries, basePat,
120  cf, baseline, sf));
121 }

123 int inflow_validate(int param, int type, int tseries, int basePat, double *cf)
124 //
125 // Purpose: Validates Inflow
126 // Input: param = -1 for Flow or Index of Pollutant
127 // type = FLOW_INFLOW, CONCEN_INFLOW or MASS_INFLOW
128 // tSeries = Time Series Index
129 // basePat = Base Pattern Index
130 // Output: cf = Unit Conversion
131 // Return: returns Error Code
132 {
133  int errcode = 0;
134  // Validate param
135  if (param >= Nobjects[POLLUT])
136  {
137  errcode = ERR_API_POLLUT_INDEX;
138  }
139  // Validate Type
140  else if (type != FLOW_INFLOW
141  && type != CONCEN_INFLOW
142  && type != MASS_INFLOW)
143  {
144  errcode = ERR_KEYWORD;
145  }
146  // Validate Timeseries Index
147  else if (tseries >= Nobjects[TSERIES])
148  {
149  errcode = ERR_API_TSERIES_INDEX;
150  }
151  // Validate Timepattern Index
152  else if (basePat >= Nobjects[TIMEPATTERN])
153  {
154  errcode = ERR_API_PATTERN_INDEX;
155  }
156  else
157  {
158  // --- assign type & cf values for a FLOW inflow
159  if ( type == FLOW_INFLOW )
160  {
161  *cf = 1.0/UCF(FLOW);
162  }
163  // --- include LperFT3 term in conversion factor for MASS_INFLOW
164  else if ( type == MASS_INFLOW )
165  {
166  *cf /= LperFT3;
167  }
168  }
169 
170  return(errcode);
171 }

174 int inflow_setExtInflow(int j, int param, int type, int tseries, int basePat,
175  double cf, double baseline, double sf)
176 // Purpose: This function assigns property values to the inflow object
177 // Inputs: j = Node index
178 // param = FLOW (-1) or pollutant index
179 // type = FLOW, CONCEN or MASS inflow
180 // tSeries = time series index
181 // basePat = baseline pattern
182 // cf = units conversion factor
183 // baseline = baseline inflow value
184 // sf = scaling factor
185 // Return: returns Error Code
186 
187 {
188  int errcode = 0;
189 
190  // Validate Inflow
191  errcode = inflow_validate(param, type, tseries, basePat, &cf);
192 
193  if (errcode == 0)
194  {
195  TExtInflow* inflow; // external inflow object
196 
197  // --- check if an external inflow object for this constituent already exists
198  inflow = Node[j].extInflow;
199  while ( inflow )
200  {
201  if ( inflow->param == param ) break;
202  inflow = inflow->next;
203  }
204 
205  // --- if it doesn't exist, then create it
206  if ( inflow == NULL )
207  {
208  inflow = (TExtInflow *) malloc(sizeof(TExtInflow));
209  if ( inflow == NULL )
210  {
211  return error_setInpError(ERR_MEMORY, "");
212  }
213  inflow->next = Node[j].extInflow;
214  Node[j].extInflow = inflow;
215  }
216 
217  // Assigning Values to the inflow object
218  inflow->param = param;
219  inflow->type = type;
220  inflow->tSeries = tseries;
221  inflow->cFactor = cf;
222  inflow->sFactor = sf;
223  inflow->baseline = baseline;
224  inflow->basePat = basePat;
225  inflow->extIfaceInflow = 0.0;
226  }
227  return(errcode);
228 }

252 double inflow_getExtInflow(TExtInflow* inflow, DateTime aDate)
253 //
254 // Input: inflow = external inflow data structure
255 // aDate = current simulation date/time
256 // Output: returns current value of external inflow parameter
257 // Purpose: retrieves the value of an external inflow at a specific
258 // date and time.
259 //
260 {
261  int month, day, hour;
262  int p = inflow->basePat; // baseline pattern
263  int k = inflow->tSeries; // time series index
264  double cf = inflow->cFactor; // units conversion factor
265  double sf = inflow->sFactor; // scaling factor
266  double blv = inflow->baseline; // baseline value
267  double tsv = 0.0; // time series value
268  double extIfaceInflow = inflow->extIfaceInflow;// external interfacing inflow
269 
270  if ( p >= 0 )
271  {
272  month = datetime_monthOfYear(aDate) - 1;
273  day = datetime_dayOfWeek(aDate) - 1;
274  hour = datetime_hourOfDay(aDate);
275  blv *= inflow_getPatternFactor(p, month, day, hour);
276  }
277  if ( k >= 0 ) tsv = table_tseriesLookup(&Tseries[k], aDate, FALSE) * sf;
278  return cf * (tsv + blv) + cf * extIfaceInflow;
279 }

283 int inflow_readDwfInflow(char* tok[], int ntoks)
284 //
285 // Input: tok[] = array of string tokens
286 // ntoks = number of tokens
287 // Output: returns an error message
288 // Purpose: reads dry weather inflow parameters from line of input data.
289 //
290 // Format of data line is:
291 // nodeID FLOW/pollutID avgValue (pattern1 pattern2 ... pattern4)
292 //
293 {
294  int i;
295  int j; // node index
296  int k; // pollutant index (-1 for flow)
297  int m; // time pattern index
298  int pats[4]; // time pattern index array
299  double x; // avg. DWF value
300  TDwfInflow* inflow; // dry weather flow inflow object
301 
302  // --- find index of node receiving the inflow
303  if ( ntoks < 3 ) return error_setInpError(ERR_ITEMS, "");
304  j = project_findObject(NODE, tok[0]);
305  if ( j < 0 ) return error_setInpError(ERR_NAME, tok[0]);
306 
307  // --- find index of inflow pollutant (-1 for FLOW)
308  k = project_findObject(POLLUT, tok[1]);
309  if ( k < 0 )
310  {
311  if ( match(tok[1], w_FLOW) ) k = -1;
312  else return error_setInpError(ERR_NAME, tok[1]);
313  }
314 
315  // --- get avg. value of DWF inflow
316  if ( !getDouble(tok[2], &x) )
317  return error_setInpError(ERR_NUMBER, tok[2]);
318  if ( k == -1 ) x /= UCF(FLOW);
319 
320  // --- get time patterns assigned to the inflow
321  for (i=0; i<4; i++) pats[i] = -1;
322  for (i=3; i<7; i++)
323  {
324  if ( i >= ntoks ) break;
325  if ( strlen(tok[i]) == 0 ) continue;
326  m = project_findObject(TIMEPATTERN, tok[i]);
327  if ( m < 0 ) return error_setInpError(ERR_NAME, tok[i]);
328  pats[i-3] = m;
329  }
330 
331  // --- check if inflow for this constituent already exists
332  inflow = Node[j].dwfInflow;
333  while ( inflow )
334  {
335  if ( inflow->param == k ) break;
336  inflow = inflow->next;
337  }
338 
339  // --- if it doesn't exist, then create it
340  if ( inflow == NULL )
341  {
342  inflow = (TDwfInflow *) malloc(sizeof(TDwfInflow));
343  if ( inflow == NULL ) return error_setInpError(ERR_MEMORY, "");
344  inflow->next = Node[j].dwfInflow;
345  Node[j].dwfInflow = inflow;
346  }
347 
348  // --- assign property values to the inflow object
349  inflow->param = k;
350  inflow->avgValue = x;
351  for (i=0; i<4; i++) inflow->patterns[i] = pats[i];
352  return 0;
353 }

377 void inflow_initDwfInflow(TDwfInflow* inflow)
378 //
379 // Input: inflow = dry weather inflow data structure
380 // Output: none
381 // Purpose: initialzes a dry weather inflow by ordering its time patterns.
382 //
383 // This function sorts the user-supplied time patterns for a dry weather
384 // inflow in the order of the PatternType enumeration (monthly, daily,
385 // weekday hourly, weekend hourly) to help speed up pattern processing.
386 //
387 {
388  int i, p;
389  int tmpPattern[4]; // index of each type of DWF pattern
390 
391  // --- assume no patterns were supplied
392  for (i=0; i<4; i++) tmpPattern[i] = -1;
393 
394  // --- assign supplied patterns to proper position (by type) in tmpPattern
395  for (i=0; i<4; i++)
396  {
397  p = inflow->patterns[i];
398  if ( p >= 0 ) tmpPattern[Pattern[p].type] = p;
399  }
400 
401  // --- re-fill inflow pattern array by pattern type
402  for (i=0; i<4; i++) inflow->patterns[i] = tmpPattern[i];
403 }

407 double inflow_getDwfInflow(TDwfInflow* inflow, int month, int day, int hour)
408 //
409 // Input: inflow = dry weather inflow data structure
410 // month = current month of year of simulation
411 // day = current day of week of simulation
412 // hour = current hour of day of simulation
413 // Output: returns value of dry weather inflow parameter
414 // Purpose: computes dry weather inflow value at a specific point in time.
415 //
416 {
417  int p1, p2; // pattern index
418  double f = 1.0; // pattern factor
419 
420  p1 = inflow->patterns[MONTHLY_PATTERN];
421  if ( p1 >= 0 ) f *= inflow_getPatternFactor(p1, month, day, hour);
422  p1 = inflow->patterns[DAILY_PATTERN];
423  if ( p1 >= 0 ) f *= inflow_getPatternFactor(p1, month, day, hour);
424  p1 = inflow->patterns[HOURLY_PATTERN];
425  p2 = inflow->patterns[WEEKEND_PATTERN];
426  if ( p2 >= 0 )
427  {
428  if ( day == 0 || day == 6 )
429  f *= inflow_getPatternFactor(p2, month, day, hour);
430  else if ( p1 >= 0 )
431  f *= inflow_getPatternFactor(p1, month, day, hour);
432  }
433  else if ( p1 >= 0 ) f *= inflow_getPatternFactor(p1, month, day, hour);
434  return f * inflow->avgValue;
435 
436 }

456 int inflow_readDwfPattern(char* tok[], int ntoks)
457 //
458 // Input: tok[] = array of string tokens
459 // ntoks = number of tokens
460 // Output: returns an error message
461 // Purpose: reads values of a time pattern from a line of input data.
462 //
463 // Format of data line is:
464 // patternID patternType value(1) value(2) ...
465 // patternID value(n) value(n+1) .... (for continuation lines)
466 {
467  int i, j, k, n = 1;
468 
469  // --- check for minimum number of tokens
470  if ( ntoks < 2 ) return error_setInpError(ERR_ITEMS, "");
471 
472  // --- check that pattern exists in database
473  j = project_findObject(TIMEPATTERN, tok[0]);
474  if ( j < 0 ) return error_setInpError(ERR_NAME, tok[0]);
475 
476  // --- check if this is first line of pattern
477  // (ID pointer will not have been assigned yet)
478  if ( Pattern[j].ID == NULL )
479  {
480  // --- assign ID pointer & pattern type
481  Pattern[j].ID = project_findID(TIMEPATTERN, tok[0]);
482  k = findmatch(tok[1], PatternTypeWords);
483  if ( k < 0 ) return error_setInpError(ERR_KEYWORD, tok[1]);
484  Pattern[j].type = k;
485  n = 2;
486  }
487 
488  // --- start reading pattern factors from rest of line
489  while ( ntoks > n && Pattern[j].count < 24 )
490  {
491  i = Pattern[j].count;
492  if ( !getDouble(tok[n], &Pattern[j].factor[i]) )
493  return error_setInpError(ERR_NUMBER, tok[n]);
494  Pattern[j].count++;
495  n++;
496  }
497  return 0;
498 }

502 double inflow_getPatternFactor(int p, int month, int day, int hour)
503 //
504 // Input: p = time pattern index
505 // month = current month of year of simulation
506 // day = current day of week of simulation
507 // hour = current hour of day of simulation
508 // Output: returns value of a time pattern multiplier
509 // Purpose: computes time pattern multiplier for a specific point in time.
510 {
511  switch ( Pattern[p].type )
512  {
513  case MONTHLY_PATTERN:
514  if ( month >= 0 && month < 12 ) return Pattern[p].factor[month];
515  break;
516  case DAILY_PATTERN:
517  if ( day >= 0 && day < 7 ) return Pattern[p].factor[day];
518  break;
519  case HOURLY_PATTERN:
520  if ( hour >= 0 && hour < 24 ) return Pattern[p].factor[hour];
521  break;
522  case WEEKEND_PATTERN:
523  if ( day == 0 || day == 6 )
524  {
525  if ( hour >= 0 && hour < 24 ) return Pattern[p].factor[hour];
526  }
527  break;
528  }
529  return 1.0;
530 }

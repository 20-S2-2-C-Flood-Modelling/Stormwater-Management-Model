94 int DLLEXPORT swmm_setSimulationDateTime(int timetype, char *dtimestr)
95 //
96 // Input: timetype = time type to return
97 // DateTime String
98 // Return: API Error
99 // Purpose: Get the simulation start, end and report date times
100 {
101  int errcode = 0;
102 
103  char theDate[10];
104  char theTime[9];
105 
106  // Check if Open
107  if(swmm_IsOpenFlag() == FALSE)
108  {
109  errcode = ERR_API_INPUTNOTOPEN;
110  }
111  // Check if Simulation is Running
112  else if(swmm_IsStartedFlag() == TRUE)
113  {
114  errcode = ERR_API_SIM_NRUNNING;
115  }
116  else
117  {
118  strncpy(theDate, dtimestr, 10);
119  strncpy(theTime, dtimestr+11, 9);
120 
121  switch(timetype)
122  {
123  //StartDateTime (globals.h)
124  case SM_STARTDATE:
125  project_readOption("START_DATE", theDate);
126  project_readOption("START_TIME", theTime);
127  StartDateTime = StartDate + StartTime;
128  TotalDuration = floor((EndDateTime - StartDateTime) * SECperDAY);
129  // --- convert total duration to milliseconds
130  TotalDuration *= 1000.0;
131  break;
132  //EndDateTime (globals.h)
133  case SM_ENDDATE:
134  project_readOption("END_DATE", theDate);
135  project_readOption("END_TIME", theTime);
136  EndDateTime = EndDate + EndTime;
137  TotalDuration = floor((EndDateTime - StartDateTime) * SECperDAY);
138  // --- convert total duration to milliseconds
139  TotalDuration *= 1000.0;
140  break;
141  //ReportStart (globals.h)
142  case SM_REPORTDATE:
143  project_readOption("REPORT_START_DATE", theDate);
144  project_readOption("REPORT_START_TIME", theTime);
145  ReportStart = ReportStartDate + ReportStartTime;
146  break;
147  default: errcode = ERR_API_OUTBOUNDS; break;
148  }
149  }
150 
151  return (errcode);
152 }
